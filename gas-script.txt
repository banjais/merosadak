/**
 * 📡 SADAK-SATHI MASTER DATA PROVIDER
 * Serves JSON data via doGet() for the backend pull
 */

const SHEET_ID = "1gfbACf6IkjNhrC_xUDuiFPUyOmgqsuU8mqKVED7Gook";
const SHEET_TAB = "Roads";

function doGet(e) {
  try {
    const ss = SpreadsheetApp.openById(SHEET_ID);
    const sheet = ss.getSheetByName(SHEET_TAB);
    
    if (!sheet) {
      return createJsonResponse({ 
        status: "Error", 
        message: `Sheet "${SHEET_TAB}" not found`
      });
    }

    // Get all data including empty rows to ensure we catch everything
    const lastRow = sheet.getLastRow();
    const lastCol = sheet.getLastColumn();
    
    if (lastRow < 2) {
      return createJsonResponse({ 
        status: "Success", 
        data: [], 
        count: 0,
        message: "No data found in sheet"
      });
    }

    const data = sheet.getRange(2, 1, lastRow - 1, lastCol).getValues();

    // Get headers from row 1
    const headers = sheet.getRange(1, 1, 1, lastCol).getValues()[0];

    // Map rows to objects
    const result = data.map((row, idx) => {
      const obj = {};
      headers.forEach((header, i) => {
        if (header && header.toString().trim()) {
          const key = header.toString().trim().toLowerCase().replace(/\s+/g, '_');
          obj[key] = (row[i] != null) ? row[i].toString().trim() : "";
        }
      });
      return obj;
    }).filter(item => {
      // Keep rows that have at least road_refno OR status
      const ref = (item.road_refno || item.road_refno || "").trim();
      const stat = (item.status || "").trim();
      return ref !== "" || stat !== "";
    });

    console.log(`📊 Rows returned: ${result.length}`);

    return createJsonResponse({
      status: "Success", 
      data: result, 
      count: result.length,
      syncedAt: new Date().toISOString()
    });

  } catch (err) {
    console.error("Error:", err.message);
    return createJsonResponse({ 
      status: "Error", 
      message: err.message 
    });
  }
}

function createJsonResponse(data) {
  return ContentService.createTextOutput(JSON.stringify(data))
    .setMimeType(ContentService.MimeType.JSON);
}
